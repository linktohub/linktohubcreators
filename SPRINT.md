```ts
export async function POST(req: NextRequest) {
  const { enabled } = await req.json();
  // ... auth check ...
  const { data: creator } = await supabase.from("creators")
    .select("stripe_account_id").eq("user_id", session.user.id).single();

  // Update Stripe account payout schedule
  await stripe.accounts.update(creator.stripe_account_id, {
    settings: {
      payouts: {
        schedule: enabled
          ? { interval: "manual" }      // creator triggers instant payouts
          : { interval: "daily", delay_days: 2 }  // standard
      }
    }
  });

  // Store preference
  await supabase.from("creators").update({ instant_payouts_enabled: enabled }).eq("user_id", session.user.id);
  return NextResponse.json({ ok: true });
}
```

```ts
// After order creation, check if buyer came via affiliate
const { data: referral } = await admin
  .from("affiliate_referrals")
  .select("referral_code, referrer_creator_id")
  .eq("referred_user_id", pi.metadata.fan_id || "")   // fan_id in PI metadata if fan logged in
  .order("created_at", { ascending: false })
  .limit(1)
  .single();

if (referral) {
  const commissionRate = 0.20;  // 20% of platform fee goes to referrer
  const commissionCents = Math.round(platformFee * commissionRate);
  await admin.from("affiliate_commissions").insert({
    order_id: order.id,
    referral_code: referral.referral_code,
    referrer_creator_id: referral.referrer_creator_id,
    commission_cents: commissionCents,
    status: "pending",
  });
```
